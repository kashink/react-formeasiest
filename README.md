## Install

```bash
npm install --save react-formeasiest
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-formeasiest'
import 'react-formeasiest/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
